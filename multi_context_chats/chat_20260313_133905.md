#!/bin/bash

# Cores para o output
GREEN='\033[0;32m'
RED='\033[0;31m'
NC='\033[0m' 

echo -e "Executando suite de validação do código...\n"
FAILS=0

function check() {
    local msg=$1
    local condition=$2
    if eval "$condition"; then
        echo -e "${GREEN}[ PASS ]${NC} $msg"
    else
        echo -e "${RED}[ FAIL ]${NC} $msg"
        FAILS=$((FAILS+1))
    fi
}

# 1. Arquivos de popup duplicados
check "Arquivo lua/multi_context/popup.lua foi deletado." "[ ! -f lua/multi_context/popup.lua ]"
check "Arquivo lua/multi_context/ui/popup.lua continua intacto." "[ -f lua/multi_context/ui/popup.lua ]"

# 2. Definições no init.vim 
check "Nenhum 'command! ... Context' está sobrando no init.vim." "! grep -q '^command!.*Context' init.vim"

# 3. Definições no init.lua
check "Comandos globais com atributos (-range, etc) inseridos no init.lua." "grep -q 'command! -range Context' lua/multi_context/init.lua"
check "Nova lógica de is_popup no ToggleWorkspaceView." "grep -q 'local is_popup =' lua/multi_context/init.lua"

# 4. Vulnerabilidade Gemini
check "Função extratora de chunks usada para o Gemini." "grep -q 'local chunks, rest = extract_text_chunks(buffer)' lua/multi_context/api_handlers.lua"
check "O reset violento (buffer = \"\") foi substituído." "! grep -A 5 'local chunks, rest' lua/multi_context/api_handlers.lua | grep -q 'buffer = \"\"'"

# 5. Segurança contra efeitos colaterais
check "Arquivo context_apis.json não foi alterado ou quebrado." "[ -f context_apis.json ] && head -n 1 context_apis.json | grep -q '{'"

echo -e "\nResultado Final:"
if[ $FAILS -eq 0 ]; then
    echo -e "${GREEN}✓ Tudo certo! Código validado com sucesso e sem side-effects mapeados.${NC}"
else
    echo -e "${RED}✗ Encontrados $FAILS erro(s) durante a validação. Revisar os logs acima.${NC}"
fi
